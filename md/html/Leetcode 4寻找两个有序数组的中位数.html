class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        
        vector<int > ans;
        
        ans.insert(ans.end(),nums1.begin(),nums1.end());
        ans.insert(ans.end(),nums2.begin(),nums2.end());
        
        sort(ans.begin(),ans.end());
        
        
        int size = ans.size();
        
        double res;
        if(size %2 ==0 )
        {
            res=(ans[size/2]+ans[size/2-1])/2.0;
        }else{
            
            res=ans[size/2];
        }
        
        return res;
        
    }
};
