class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int ans=0,left=-1;
        //（元素，位置）
        map<int,int> m;
        for(int i=0;i<s.size();++i)
        {	//维护一个滑动窗口，【left】【窗口】，left不属于窗口，紧挨窗口最左侧
            if(m.count(s[i]) && (m[s[i]] > left))
            {	
                left=m[s[i]];
            }
            
            m[s[i]] = i;
            
		//更新ans
            ans=max(ans,i-left);
            
        }
        
        return ans;
    }
};
